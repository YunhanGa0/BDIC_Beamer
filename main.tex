\section{Introduction}
\begin{frame}{Introduction}
    \begin{itemize}
        \item Briefly introduce the topic and why it matters.
        \item Summarize the background in two or three points.
        \item State the purpose of this presentation.
    \end{itemize}
\end{frame}

\section{Main Part}
\begin{frame}{Main Part}
    \begin{itemize}
        \item Present your key argument, method, or design.
        \item Use short bullet points and place figures or tables as needed.
        \item Keep one clear message per slide.
    \end{itemize}
\end{frame}

\section{Conclusion}
\begin{frame}{Conclusion}
    \begin{itemize}
        \item Recap the most important takeaway.
        \item Highlight implications, next steps, or recommendations.
        \item Leave one final message for the audience.
    \end{itemize}
\end{frame}
